51

319void escrever_cabecalho_no_arquivo(FILE *arquivo_comprimido, int bits_de_lixo, int tamanho_arvore, Arvore *arvore)
320{
321 //setando os bits
322 uint16_t tamanho_lixo_e_arvore = 0, mascara;
323 //guarda o lixo no vetor para ser gravado no arquivo
324 tamanho_lixo_e_arvore |= bits_de_lixo << 13;
325 //gravando o tamanho da arvore no vetor
326 for(int i = 0; i < 13; i++)
327 {
328 mascara = 1 << i;
329 if(tamanho_arvore & mascara)
330 {
331 tamanho_lixo_e_arvore |= 1 << i;
332 }
333 else
334 {
335 tamanho_lixo_e_arvore |= 0 << i;
336 }
337 }
338 //Como a ordem em que a fwrite escreve depende da arquitetura do processador, ou seja, little-endian ou big-endian precisamos garantir que
339 //os dois bytes para lixo e para tamanho da arvore sejam escritos corretamente, para isso usamos a funcao htons da biblioteca arpa/inet.h
340 tamanho_lixo_e_arvore = htons(tamanho_lixo_e_arvore);
341 //gravando o lixo e o tamanho da arvore
342 fwrite(&tamanho_lixo_e_arvore, 1, 2, arquivo_comprimido);
343 //gravando a arvore
344 escrever_arvore_no_cabecalho(arquivo_comprimido, arvore);
345}

359void escrever_bits_compactados(FILE *arquivo_comprimido, uint8_t *dados, uint8_t **dicionario, long tamanho_arquivo)
360{
361 long indice_bit = 7;
362 uint8_t byte = 0;
363 for(int i = 0; i < tamanho_arquivo; i++)
364 {
365 for(int j = 0; dicionario[dados[i]][j] != '\0'; j++)
366 {
367 if(dicionario[dados[i]][j] == '1')
368 {
369 byte |= 1 << indice_bit;
370 }
371 indice_bit--;
372 if(indice_bit < 0)
373 {
374 fwrite(&byte, sizeof(unsigned char), 1, arquivo_comprimido);
375 indice_bit = 7;
376 byte = 0;
377 }
378 }
379 }
380 if(indice_bit != 7)
381 {
382 fwrite(&byte, sizeof(unsigned char), 1, arquivo_comprimido);
383 }
384
385 return;
386}

428void comprimir(char *nome_arquivo)
429{
430 //cria a fila de frequências
431 Arvore *fila = NULL;
432 //criando arvore de huffman
433 Arvore *arvore_huffman = NULL;
434 //criando a variavel do dicionario
435 uint8_t **dicionario;
436 //criando a variavel auxiliar para preencher o dicionario
437 uint8_t *aux;
438
439 //abrindo o arquivo
440 FILE *arquivo = fopen(nome_arquivo, "rb"), *arquivo_comprimido;
441 if(arquivo == NULL)
442 {
443 printf("\nArquivo não encontrado!\n");
444 exit(1);
445 }
446
447 //criando a variavel que vai manter todos os bytes
448 uint8_t *dados;
449 //criando a tabela de frequencia
450 long frequencia[Max_table], i;
451
452 //lendo os bytes do arquivo
453 //procura o fim do arquivo
454 fseek(arquivo, 0, SEEK_END);
455 //pega o tamanho do arquivo
456 long tamanho_arquivo = ftell(arquivo);
457 //volta o ponteiro para o inicio do arquivo
458 fseek(arquivo, 0, SEEK_SET);
459 //aloca memoria para a variavel dados que usaremos para pegar todos os bytes do arquivo
460 dados = malloc(tamanho_arquivo);
461 if(dados == NULL)
462 {
463 printf("\nNão foi possível alocar memória para o vetor do arquivo\n");
464 exit(1);
465 }
466 //setando os bytes para 0
467 memset(dados, 0, tamanho_arquivo);
468 //pegando todos os bytes do arquivo
469 fread(dados, 1, tamanho_arquivo, arquivo);
470 //fechando o arquivo
471 fclose(arquivo);
472
473 //iniciando as frequencias como 0
474 memset(frequencia, 0, Max_table*sizeof(long));
475
476 //obtendo frequencias dos bytes
477 for(i = 0; i < tamanho_arquivo; i++)
478 {
479 //pegando a frequencia de cada byte
480 frequencia[dados[i]]++;
481 }
482
483 //montando a lista de frequência
484 for(i = 0; i < Max_table; i++)
485 {
486 //avaliando se a frequencia e igual a 0 para nao pegarmos o byte que nao tem no arquivo
487 if(frequencia[i] != 0)
488 {
489 //fazendo o cast da variavel i para do tipo unsigned char
490 (uint8_t)i;
491 //insere na fila de frequencia de maneira organizada
492 enfileirar(&fila, novo_no_arvore(&i, frequencia[i]));
493 }
494 }
495
496 //criando a arvore de huffman
497 criar_arvore_huffman(&fila, &arvore_huffman);
498
499 //pegando a altura da arvore
500 long altura_da_arvore = altura_arvore(arvore_huffman);
501 //pegando o tamanho da arvore
502 long tamanho_da_arvore = tamanho_arvore(arvore_huffman);
503
504
505 //criando o dicionário
506 dicionario = criar_dicionario(altura_da_arvore + 1);
507
508 //criando o vetor auxiliar para preencher o dicionario
509 aux = (uint8_t*)malloc(sizeof(uint8_t) * (altura_da_arvore + 1));
510 if(aux == NULL)
511 {
512 printf("\nNão foi possível alocar memória para o array auxiliar\n");
513 exit(1);
514 }
515
516 //preenchendo o dicionario
517 gerar_codigos(dicionario, arvore_huffman, 0, aux);
518
519 //calculo do lixo de bits
520 int bits_de_lixo = lixo(dicionario, frequencia);
521 //mudanca do nome, exemplo: arquivo.txt vira arquivo.txt.huff
522 for(i = 0; i != 106; i++)
523 {
524 if(nome_arquivo[i] == '\0' && i + 6 <= 106)
525 {
526 nome_arquivo[i] = '.';
527 nome_arquivo[i + 1] = 'h';
528 nome_arquivo[i + 2] = 'u';
529 nome_arquivo[i + 3] = 'f';
530 nome_arquivo[i + 4] = 'f';
531 nome_arquivo[i + 5] = '\0';
532 break;
533 }
534 }
535
536 //escrevendo o arquivo comprimido
537 arquivo_comprimido = fopen(nome_arquivo,"wb");
538 if(arquivo_comprimido == NULL)
539 {
540 printf("\nNão foi possível abrir o arquivo de saída\n");
541 exit(1);
542 }
543 //escrevendo o cabecalho
544 escrever_cabecalho_no_arquivo(arquivo_comprimido, bits_de_lixo, tamanho_da_arvore, arvore_huffman);
545 //escrevendo os bytes compactados
546 escrever_bits_compactados(arquivo_comprimido, dados, dicionario, tamanho_arquivo);
547 //fechando o arquivo
548 fclose(arquivo_comprimido);
549
550 printf("\nArquivo comprimido com sucesso!!!\n");
551
552 //libera a memoria do dicionario
553 free_dicionario(dicionario);
554 //libera a memoria da arvore
555 free_arvore_huffman(arvore_huffman);
556 //libera a memoria do vetor auxiliar
557 free(aux);
558 //libera a memoria alocada para o vetor dados
559 free(dados);
560 aux = NULL;
561 dicionario = NULL;
562 arvore_huffman = NULL;
563 dados = NULL;
564
565 return;
