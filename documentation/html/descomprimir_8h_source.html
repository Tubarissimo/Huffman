152{
153 Arvore_D* aux = arvore;
154
155 while(i < tamanho_arquivo)
156 {
157 uint8_t byte = dados[i];
158
159 for(int j = 7; j >= 0; j--)
160 {
161 if(i == tamanho_arquivo - 1 && j < lixo) return;
162

164 {
165 aux = aux->direita;
